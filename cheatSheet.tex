\documentclass{article}
\usepackage[most]{tcolorbox}
\usepackage{xcolor}
\usepackage[margin=1in]{geometry}
\usepackage{multicol}

% Color Definitions
\definecolor{headerDark}{HTML}{333333}
\definecolor{mintBox}{HTML}{A7E8D8}
\definecolor{codeBlack}{HTML}{222222}
\definecolor{mintShadow}{HTML}{7ABAA9}

\newtcolorbox{gitcard}[1]{
    enhanced,
    sharp corners,
    colback=white,
    colframe=headerDark,
    fonttitle=\sffamily\bfseries\large,
    title={#1},
    attach boxed title to top left={yshift=-2.5mm, xshift=2mm},
    boxed title style={sharp corners, colback=headerDark, colframe=headerDark},
    left=0mm, right=2mm, top=4mm, bottom=2mm,
    boxrule=1.5pt
}

\newtcolorbox{gitsection}[1]{
    enhanced,
    sharp corners,
    colback=mintBox,
    colframe=mintBox,
    fontupper=\sffamily\bfseries,
    left=4mm, right=4mm, top=2mm, bottom=2mm,
    boxrule=0pt,
    title={#1},
    coltitle=black,
    attach title to upper,
    after title={\par\smallskip},
    grow to left by=4mm,
    % grow to right by=0mm
    % the ribbon fold corner
    finish={
        \fill[mintShadow] ([xshift=2.47mm]frame.north west) -- (frame.north west) -- ([xshift=2.47mm, yshift=1.2mm]frame.north west) -- cycle;
    }
}

\newcommand{\gitcommand}[1]{%
    \begin{tcolorbox}[
        sharp corners, 
        colback=codeBlack, 
        colframe=codeBlack, 
        boxrule=0pt,
        top=1.5mm, bottom=1.5mm, left=2mm,
        fontupper=\ttfamily\color{white}
    ]
    #1
    \end{tcolorbox}
}

\begin{document}
\setlength{\columnsep}{1cm}

\begin{multicols}{2}
\begin{gitcard}{Creating Repository}
    
    \begin{gitsection}{Download an existing repository}
        \gitcommand{git clone <url>}
    \end{gitsection}
    
    \vspace{2mm}
    
    \begin{gitsection}{Create new local repository}
        \gitcommand{git init}
        \begin{minipage}{0.3\textwidth}
            \gitcommand{<project name>}
        \end{minipage}
        \hspace{2mm}
        \begin{minipage}{0.6\textwidth}
            \textsf{\small if added: creates .git in sub-dir}
        \end{minipage}
    \end{gitsection}
    
    \vspace{2mm}
    
    \begin{gitsection}{Links the remote repo to your local one}
        \gitcommand{git remote add origin <url>}
    \end{gitsection}

\end{gitcard}

\columnbreak 

    \begin{gitcard}{Remote Work}
        \begin{gitsection}{Push Changes}
            \gitcommand{git push origin main}
        \end{gitsection}
        
        \vspace{6mm}
        
        \begin{gitsection}{Pull Changes}
            \gitcommand{git pull}
        \end{gitsection}
    \end{gitcard}

\end{multicols}
\end{document}
